            const statusEl = document.getElementById('prodModalStatus');

                statusEl.classList.remove('hidden', 'text-yellow-400', 'text-red-500');

            // Verifica estado de login na sessão (para persistir o mock)

                handleLoginSuccess();
            } else {
                handleLogout();
            }
            
            // Inicia na seção dashboard

        // --- RENDERING ---

        function renderProducts(currentProducts) {
            ui.tables.products.innerHTML = '';
            if(currentProducts.length === 0) {
                ui.tables.products.innerHTML = '<tr><td colspan="5" class="p-8 text-center text-slate-500">Nenhum produto cadastrado.</td></tr>';

            currentProducts.forEach(p => {

                tr.className = "hover:bg-slate-800/30 transition-colors border-b border-slate-800/50";
                
                let catColor = "bg-slate-700 text-slate-300";
                if(p.genre === 'console') catColor = "bg-purple-900/30 text-purple-400 border border-purple-500/20";

                if(p.genre === 'acessorios') catColor = "bg-red-900/30 text-red-400 border border-red-500/20";
                
                const formattedPrice = new Intl.NumberFormat('pt-BR', {
                    style: 'currency',
                    currency: 'BRL'
                }).format(p.price || 0);

                tr.innerHTML = `
                    <td class="p-4">
                        <div class="w-12 h-12 rounded bg-slate-800 border border-slate-700 flex items-center justify-center overflow-hidden">

                        </button>
                    </td>

                ui.tables.products.appendChild(tr);
            });
        }
